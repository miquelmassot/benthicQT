  ufEllipsoids.cpp
  ufMapGrid.cpp
  ufMapGrids.cpp

  ufEllipsoids.h
  ufMapGrid.h
  ufMapGrids.h
  ufRedfearn.h
  ufLocalRedfearn.h
  ufVincenty.h
)

#install(FILES ${${KIT}_INCS} DESTINATION ${UF_INSTALL_INCLUDE_DIR}/${KIT})
